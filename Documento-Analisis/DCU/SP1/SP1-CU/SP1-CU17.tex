%ASIGNAR TAREAS: NAIBI.
\begin{UseCase}{SP1-CU17}{Asignar Tareas}{El actor podrá asignar tareas para el registro de Unidades de Aprendizaje a distintos recursos humanos.}
        \UCitem{Versión}{\color{Gray}1.1}
        \UCitem{Autor}{\color{Gray}Ramírez Martínez Janet Naibi}
        \UCitem{Supervisa}{\color{Gray}Ramírez Martínez Janet Naibi}
        \UCitem{Actor}{\hyperlink{JIE}{Jefe de Innovación Educativa}}
        \UCitem{Propósito}{Poder repartir el trabajo de registro de la Unidad de Aprendizaje.}
        \UCitem{Origen}{Mouse}
        \UCitem{Salidas}{
            \begin{itemize}
                \item \MSGref{MSG25}{Servicios no disponibles.}
                \item \MSGref{MSG29}{¿Está seguro que desea cancelar? Se perderán todos los avances sin guardar.}
                \item \MSGref{MSG63}{¿Desea asignar la tarea a otro usuario?}
                \item \MSGref{MSG66}{Esta tarea ya fue asignada.}
            \end{itemize}
        }
        \UCitem{Destino}{Pantalla.}
        \UCitem{Precondiciones}{Se deben haber registrado las unidades de aprendizaje que se quieren asignar a algún recurso humano}.
        \UCitem{Postcondiciones}{El usuario al que se le asignó la tarea podrá ver la tarea en su menú correspondiente y elaborarla.}
        \UCitem{Errores}{}
        \UCitem{Estado}{Revisión.}
        \UCitem{Observaciones}{}
\end{UseCase}

%--------------------------- CU TRAYECTORIA PRINCIPAL -------------------------
\begin{UCtrayectoria}{Principal}
    \UCpaso[\UCactor] Presiona el botón \BtnTask de la interfaz de usuario \IUref{CMC-J}{Consultar mapa curricular}.
    \UCpaso Muestra la interfaz de usuario \IUref{AUA}{Asignar Unidades de Aprendizaje}.
    \UCpaso[\UCactor] Selecciona el recurso humano al que le quiere asignar la elaboración de la Unidad de Aprendizaje.
    \UCpaso[\UCactor] Termina la operación presionando el botón \IUbutton{Asignar}. [Trayectoria A] [Trayectoria B] [Trayectoria C]
    \UCpaso Muestra el mensaje \MSGref{MSG63}{¿Desea asignar la tarea a otro usuario?}.
    \UCpaso[\UCactor] Presiona el botón \IUbutton{No}. [Trayectoria D]
    \UCpaso Muestra la interfaz de usuario \IUref{UAU}{Asignar Unidades de Aprendizaje}.
\end{UCtrayectoria}

%------------------------ CU TRAYECTORIA ALTERNATIVA A -------------------------

\begin{UCtrayectoriaA}{A}{El actor selecciona a un recurso humano que ya ha sido asignado a esa tarea previamente.}
    \UCpaso Muestra el mensaje \MSGref{MSG66}{Esta tarea ya fue asignada}.
    \UCpaso[\UCactor] Cierra el mensaje presionando el botón \IUbutton{Aceptar}.
    \UCpaso Muestra la interfaz de usuario \IUref{UAU}{Asignar Unidades de Aprendizaje}.
    \UCpaso Continua en el paso 3 de la trayectoria principal del \UCref{SP1-CU17}.
\end{UCtrayectoriaA}

%------------------------ CU TRAYECTORIA ALTERNATIVA B -------------------------

\begin{UCtrayectoriaA}{B}{El actor desea cancelar la asignación de recursos humanos a una tarea.}
    \UCpaso[\UCactor] Presiona el botón \IUbutton{Cancelar}
    \UCpaso Muestra el mensaje \MSGref{MSG29}{¿Está seguro que desea cancelar? Se perderán todos los avances sin guardar.}
    \UCpaso[\UCactor] Presiona el botón \IUbutton{Si}. [Trayectoria B.1]
    \UCpaso Cierra el mensaje.
    \UCpaso Muestra la interfaz de usuario \IUref{CMC-J}{Consultar mapa curricular}.
\end{UCtrayectoriaA}

%------------------------ CU TRAYECTORIA ALTERNATIVA B.1 -------------------------

\begin{UCtrayectoriaA}{B.1}{El actor no desea cancelar la asignación de recursos humanos a una tarea.}
    \UCpaso[\UCactor] Presiona el botón \IUbutton{No}.
    \UCpaso Cierra el mensaje.
    \UCpaso Muestra la interfaz de usuario \IUref{UAU}{Asignar Unidades de Aprendizaje}.
    \UCpaso Continua en el paso 3 de la trayectoria principal del \UCref{SP1-CU17}
\end{UCtrayectoriaA}

%------------------------ CU TRAYECTORIA ALTERNATIVA C -------------------------

\begin{UCtrayectoriaA}{C}{El actor no selecciona ningún recurso humano y quiere finalizar.}
    \UCpaso[\UCactor] No selecciona ningún recurso humano.
    \UCpaso[\UCactor] Presiona el botón \IUbutton{Asignar}.
    \UCpaso Muestra el mensaje \MSGref{MSG25}{Servicios no disponibles.}
    \UCpaso Muestra la interfaz de usuario \IUref{UAU}{Asignar Unidades de Aprendizaje}.
    \UCpaso Continua en el paso 3 de la trayectoria principal del \UCref{SP1-CU17}
\end{UCtrayectoriaA}

%------------------------ CU TRAYECTORIA ALTERNATIVA D -------------------------

\begin{UCtrayectoriaA}{D}{El actor quiere asignar a otro recurso humano para la elaboración de la Unidad de Aprendizaje.}

    \UCpaso Cierra el mensaje.
    \UCpaso Muestra la \IUref{UAU}{Asignar Unidades de Aprendizaje}.
