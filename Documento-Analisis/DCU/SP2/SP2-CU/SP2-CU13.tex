%--------------------------- CU TRAYECTORIA PRINCIPAL -------------------------
\begin{UCtrayectoria}{Principal}


    \UCpaso[\UCactor] Presiona el botón \IUbutton{Eliminar comentario}. 
    
    \UCpaso Muestra el mensaje \MSGref{MSG48}{¿Está seguro que desea eliminar el comentario?}.

    \UCpaso[\UCactor] Cierra el mensaje presionando \IUbutton{Aceptar}. \hyperlink{SP2-CU13-A}{Trayectoria A}.
    
    \UCpaso Elimina el comentario del sistema. \hyperlink{SP2-CU13-B}{Trayectoria B}.
    
    \UCpaso Muestra el \MSGref{MSG49}{Se ha eliminado el comentario.}

    \UCpaso[\UCactor] Cierra el mensaje presionando \IUbutton{Aceptar}.

     \UCpaso Se muestra la pantalla que invocó al SP2-CU13 sin el nuevo comentario que se eliminó en la bitácora de comentarios. 


\end{UCtrayectoria}
