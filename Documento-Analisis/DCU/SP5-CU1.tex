% INICIAR SESION
\begin{UseCase}{SP5-CU1}{Iniciar sesión}{Los diversos usuarios podrán ingresar al sistema.}
        \UCitem{Versión}{\color{Gray}2.0}
        \UCitem{Autor}{\color{Gray}Nicolás Sayago Abigail}
     
        \UCitem{Supervisa}{\color{Gray}Ramos Díaz Enrique}
        \UCitem{Actor}{\hyperlink{JDIC}{JDIC}, \hyperlink{JDIA}{JDIA}, \hyperlink{Analista de la DES}{Analista de la DES},\hyperlink{JIE}{JIE}, \hyperlink{Analista de la UA}{Analista de la UA}, \hyperlink{Docente}{Docente}. }
        \UCitem{Propósito}{Contribuir a la gestión del proceso de creación de Unidades de Aprendizaje.}
        \UCitem{Entradas}{Las entradas para el inicio de sesión serán:
          \begin{itemize}
            \item Correo electrónico.
            \item Contraseña.
          \end{itemize}
        }
        \UCitem{Origen}{Teclado.}
        \UCitem{Salidas}{
            \begin{itemize}
                \item Página principal del sistema.
                \item \MSGref{MSG0}{Correo y/o contraseña no válidas.}
                
                \item \MSGref{MSG25}{Servicios no disponibles}.
                \item \MSGref{MSG26}{El correo ingresado no se encuentra registrado en el sistema.}
            \end{itemize}
        }
        \UCitem{Destino}{Pantalla.}
        \UCitem{Precondiciones}{1. El correo electrónico ingresado tiene una cuenta de usuario en el sistema.
        }
        \UCitem{Postcondiciones}{1. El usuario puede ver sus opciones en el sistema, dependiendo del caso.}
        \UCitem{Errores}{
            \begin{itemize}
                \item El correo ingresado no se encuentra registrado en el sistema.
                \item Correo electrónico incorrecto.
                \item Contraseña incorrecta.
                \item Correo electrónico y contraseña incorrecta.
                \item Hubo un problema al conectarse con el servidor.
                \item Hubo un problema al conectarse con la base de datos.
            \end{itemize}
        }
        \UCitem{Estado}{Acceso.}
        \UCitem{Observaciones}{}
\end{UseCase}

%--------------------------- CU TRAYECTORIA PRINCIPAL -------------------------
\begin{UCtrayectoria}{Principal}
    
    \UCpaso[\UCactor] Ingresa el url en el navegador establecido en la \BRref{BR37}{Dirección web del sistema}. [Trayectoria A] 
    \UCpaso Muestra la interfaz de usuario \IUref{IU3}{Inicio de sesion}.
    \UCpaso[\UCactor] Ingresa correo electrónico.
    \UCpaso[\UCactor] Ingresa su contraseña.
    \UCpaso[\UCactor] Presiona el botón \IUbutton{Entrar} de la interfaz de usuario \IUref{IU3}{Inicio de sesión}  [Trayectoria A][Trayectoria C].
    \UCpaso Muestra la interfaz de la pantalla principal.[Trayectoria D] [Trayectoria E] [Trayectoria F] [Trayectoria G]
\end{UCtrayectoria}
